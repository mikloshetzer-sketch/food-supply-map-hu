// VASÚTÁLLOMÁSOK BETÖLTÉSE let railStationsLayer = null; fetch('data/rail_stations.geojson') .then(res => res.json()) .then(data => { railStationsLayer = L.geoJSON(data, { pointToLayer: function(feature, latlng) { return L.circleMarker(latlng, { radius: 5, color: "#0033cc", weight: 1, fillColor: "#0066ff", fillOpacity: 0.9 }); }, onEachFeature: function(feature, layer) { layer.bindPopup("Vasútállomás"); } }); });
